function doPost(e) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var data = JSON.parse(e.postData.contents);
  var result = {status:"ok"};

  // Log play to Plays sheet
  if (data.action === "add" || data.action === "delete") {
    var playsSheet = ss.getSheetByName("Plays");
    if (!playsSheet) {
      playsSheet = ss.insertSheet("Plays");
      playsSheet.appendRow(["Date","Game","Winner","Notes","J Rating","E Rating"]);
    }
    if (data.action === "add") {
      playsSheet.appendRow([data.date, data.game, data.winner, data.notes || "", data.jRat || "", data.eRat || ""]);
      // Mark game as Played + write ratings to Games sheet
      if (data.markPlayed || data.jRat || data.eRat) {
        var gamesSheet = ss.getSheetByName("Games");
        if (gamesSheet) {
          var gv = gamesSheet.getDataRange().getValues();
          var headers = gv[0];
          // Find columns by name — robust to column order changes
          var playedCol  = headers.indexOf("Played")   + 1; // col J = 10
          var jRatCol    = headers.indexOf("J Rating")  + 1; // col K = 11
          var eRatCol    = headers.indexOf("E Rating")  + 1; // col L = 12
          for (var i = 1; i < gv.length; i++) {
            if (gv[i][0] === data.game) {
              if (data.markPlayed && playedCol > 0) {
                var cur = (gv[i][playedCol-1]||'').toString().toLowerCase();
                if (cur !== 'yes') gamesSheet.getRange(i+1, playedCol).setValue("Yes");
              }
              if (data.jRat && jRatCol > 0) {
                gamesSheet.getRange(i+1, jRatCol).setValue(parseFloat(data.jRat));
              }
              if (data.eRat && eRatCol > 0) {
                gamesSheet.getRange(i+1, eRatCol).setValue(parseFloat(data.eRat));
              }
              break;
            }
          }
        }
      }
    } else if (data.action === "delete") {
      result.playDeleted = false;
      var v = playsSheet.getDataRange().getValues();
      var playTimeZone = ss.getSpreadsheetTimeZone() || Session.getScriptTimeZone();
      for (var i = v.length - 1; i >= 1; i--) {
        if (normalizePlayDate_(v[i][0], playTimeZone) === normalizePlayDate_(data.date, playTimeZone) &&
            normalizePlayText_(v[i][1]) === normalizePlayText_(data.game) &&
            normalizePlayText_(v[i][2]) === normalizePlayText_(data.winner)) {
          playsSheet.deleteRow(i + 1);
          result.playDeleted = true;
          break;
        }
      }
      if (!result.playDeleted) result.message = "No matching play was found in the Plays sheet.";
    }
  }

  // Update ratings without logging a play
  if (data.action === "rate") {
    result.ratingsUpdated = false;
    var gamesSheet = ss.getSheetByName("Games");
    if (gamesSheet) {
      var gv = gamesSheet.getDataRange().getValues();
      var headers = gv[0];
      var jRatCol = headers.indexOf("J Rating") + 1;
      var eRatCol = headers.indexOf("E Rating") + 1;
      for (var i = 1; i < gv.length; i++) {
        if ((gv[i][0] || "").toString().trim() === (data.game || "").toString().trim()) {
          if (data.jRat !== "" && jRatCol > 0) gamesSheet.getRange(i + 1, jRatCol).setValue(parseFloat(data.jRat));
          if (data.eRat !== "" && eRatCol > 0) gamesSheet.getRange(i + 1, eRatCol).setValue(parseFloat(data.eRat));
          result.ratingsUpdated = true;
          break;
        }
      }
    }
  }

  // Update Excluded column (found by header name) in Games sheet
  if (data.action === "exclude") {
    var gamesSheet = ss.getSheetByName("Games");
    if (gamesSheet) {
      var gv = gamesSheet.getDataRange().getValues();
      // Find or create the Excluded column
      var headers = gv[0];
      var exclCol = headers.indexOf('Excluded') + 1; // 1-based
      if (exclCol === 0) {
        // Column doesn't exist yet - add it after last column
        exclCol = gv[0].length + 1;
        gamesSheet.getRange(1, exclCol).setValue('Excluded');
      }
      for (var i = 1; i < gv.length; i++) {
        if (gv[i][0] === data.game) {
          gamesSheet.getRange(i + 1, exclCol).setValue(data.excluded ? 1 : '');
          break;
        }
      }
    }
  }

  return ContentService.createTextOutput(JSON.stringify(result)).setMimeType(ContentService.MimeType.JSON);
}
function doGet(e) {
  return ContentService.createTextOutput(JSON.stringify({status:"ok"})).setMimeType(ContentService.MimeType.JSON);
}
function normalizePlayDate_(value, timeZone) {
  if (value instanceof Date && !isNaN(value.getTime())) {
    return Utilities.formatDate(value, timeZone, "yyyy-MM-dd");
  }
  var text = String(value || "").trim();
  var iso = text.match(/^(\d{4})-(\d{2})-(\d{2})/);
  if (iso) return iso[1] + "-" + iso[2] + "-" + iso[3];
  var slash = text.match(/^(\d{1,2})\/(\d{1,2})\/(\d{4})$/);
  if (slash) return slash[3] + "-" + ("0" + slash[1]).slice(-2) + "-" + ("0" + slash[2]).slice(-2);
  var parsed = new Date(text);
  return isNaN(parsed.getTime()) ? text : Utilities.formatDate(parsed, timeZone, "yyyy-MM-dd");
}
function normalizePlayText_(value) {
  return String(value || "").trim().toLowerCase();
}

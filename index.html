How It Works

Connect this dashboard to a Google Sheet so your game list and play log stay in sync. Add a new game to the sheet → refresh → it appears here. Log a play here → it writes back to the sheet.

Setup — your existing spreadsheet works as-is, no column changes needed:
1. Open board_games_updated in Google Sheets (already in your Drive)
2. Go to File → Share → Publish to web
3. First dropdown: select the Games tab · Second: Comma-separated values (.csv) → Publish → copy that URL
4. Add a new tab named Plays with four columns in row 1: Date | Game | Winner | Notes | J Rating | E Rating
5. Back in Publish to web — publish the Plays tab as CSV too → copy that URL
6. Paste both URLs into the fields below and click Save & Sync

Column mapping (your exact column names, no changes needed):
Game · Shelf · Players (Manufacturer) · Best At · Plays well at 2? - from chat · Good solo option · BGG Score · BGG Category · Played · Rating · Expansion · E knows? · Potentially sell? · Notes

▸ Apps Script for Writing Plays Back to Sheets (click to expand)

In your Google Sheet, go to Extensions → Apps Script, paste this, then Deploy → New deployment → Web app (Execute as: Me, Access: Anyone):

function doPost(e) {
  var ss = SpreadsheetApp.getActiveSpreadsheet();
  var data = JSON.parse(e.postData.contents);

  // Log play to Plays sheet
  if (data.action === "add" || data.action === "delete") {
    var playsSheet = ss.getSheetByName("Plays");
    if (!playsSheet) {
      playsSheet = ss.insertSheet("Plays");
      playsSheet.appendRow(["Date","Game","Winner","Notes","J Rating","E Rating"]);
    }
    if (data.action === "add") {
      playsSheet.appendRow([data.date, data.game, data.winner, data.notes || "", data.jRat || "", data.eRat || ""]);
      // Mark game as Played + write ratings to Games sheet
      if (data.markPlayed || data.jRat || data.eRat) {
        var gamesSheet = ss.getSheetByName("Games");
        if (gamesSheet) {
          var gv = gamesSheet.getDataRange().getValues();
          var headers = gv[0];
          // Find columns by name — robust to column order changes
          var playedCol  = headers.indexOf("Played")   + 1; // col J = 10
          var jRatCol    = headers.indexOf("J Rating")  + 1; // col K = 11
          var eRatCol    = headers.indexOf("E Rating")  + 1; // col L = 12
          for (var i = 1; i < gv.length; i++) {
            if (gv[i][0] === data.game) {
              if (data.markPlayed && playedCol > 0) {
                var cur = (gv[i][playedCol-1]||'').toString().toLowerCase();
                if (cur !== 'yes') gamesSheet.getRange(i+1, playedCol).setValue("Yes");
              }
              if (data.jRat && jRatCol > 0) {
                gamesSheet.getRange(i+1, jRatCol).setValue(parseFloat(data.jRat));
              }
              if (data.eRat && eRatCol > 0) {
                gamesSheet.getRange(i+1, eRatCol).setValue(parseFloat(data.eRat));
              }
              break;
            }
          }
        }
      }
    } else if (data.action === "delete") {
      var v = playsSheet.getDataRange().getValues();
      for (var i = v.length - 1; i >= 1; i--) {
        if (v[i][0]==data.date && v[i][1]==data.game && v[i][2]==data.winner) {
          playsSheet.deleteRow(i + 1); break;
        }
      }
    }
  }

  // Update Excluded column (found by header name) in Games sheet
  if (data.action === "exclude") {
    var gamesSheet = ss.getSheetByName("Games");
    if (gamesSheet) {
      var gv = gamesSheet.getDataRange().getValues();
      // Find or create the Excluded column
      var headers = gv[0];
      var exclCol = headers.indexOf('Excluded') + 1; // 1-based
      if (exclCol === 0) {
        // Column doesn't exist yet - add it after last column
        exclCol = gv[0].length + 1;
        gamesSheet.getRange(1, exclCol).setValue('Excluded');
      }
      for (var i = 1; i < gv.length; i++) {
        if (gv[i][0] === data.game) {
          gamesSheet.getRange(i + 1, exclCol).setValue(data.excluded ? 1 : '');
          break;
        }
      }
    }
  }

  return ContentService.createTextOutput(JSON.stringify({status:"ok"})).setMimeType(ContentService.MimeType.JSON);
}
function doGet(e) {
  return ContentService.createTextOutput(JSON.stringify({status:"ok"})).setMimeType(ContentService.MimeType.JSON);
}
